def demander_nom():
    reponse_nom = ""
    while reponse_nom == "":
        nom = input("Quel est votre nom ? ")
        reponse_nom = nom
    return reponse_nom


def demander_age(): 
    age_int = 0 
    while age_int == 0:
        age_str = input("Quelle est votre âge ? ")
        try:
            age_int = int(age_str)  
        except:
            print("ERREUR: Vous devez rentrer un nombre pour l'âge")
    return age_int
